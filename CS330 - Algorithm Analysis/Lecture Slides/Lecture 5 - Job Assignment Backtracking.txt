#include <iostream>
#include <vector>
using namespace std;

int n=4, cost=0;
int x[4], c[4][4] = { 9,2,7,8,
                      6,4,3,7,
                      5,8,1,8,
                      7,6,9,4 };

void work(int i, int count, vector<int> & solution) {
  if (i+1>n) {
    cost = count;
    for (int i = 0; i < solution.size(); ++i) { std::cout << solution[i] << " "; }
    cout << "cost:" << cost << endl;    
    return;
  }

  for (int j = 0; j < n; j++) {
    if (x[j] == 0) {
      x[j] = 1;
      solution.push_back(j);
      work(i + 1, count + c[i][j],solution);
      x[j] = 0;
      solution.pop_back();
    }
  }
}

int main() {
  vector<int> solution;
  work(0, 0, solution);

  return 0;
}
